// Code block example
function toggleTheme() {
  const theme = document.documentElement.getAttribute('data-theme');
  const newTheme = theme === 'dark' ? 'light' : 'dark';
  document.documentElement.setAttribute('data-theme', newTheme);
}

Theme Toggle Instructions

You can toggle between light and dark themes using:

  • The theme toggle button in the site header
  • Keyboard shortcut: Ctrl+Shift+T (or Cmd+Shift+T on Mac)

Your theme preference will be automatically saved and restored when you visit the site again.
